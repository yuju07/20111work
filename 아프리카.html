# 평균 득점 설정

  • lambda_cv = 75 # 카보베르데 평균 득점
  • lambda_ng = 85 # 나이지리아 평균 득점

    # 시뮬레이션 횟수 설정

  • num_simulations = 10000

    # 시뮬레이션 실행

  • cv_wins = 0
  • ng_wins = 0
  • for _ in range(num_simulations):
  • cv_score = np.random.poisson(lambda_cv)
  • ng_score = np.random.poisson(lambda_ng)
  • if cv_score > ng_score:
  • cv_wins += 1
  • elif ng_score > cv_score:
  • ng_wins += 1

    # 승리 확률 계산

  • cv_win_prob = cv_wins / num_simulations
  • ng_win_prob = ng_wins / num_simulations

  • 카보베르데 승리 확률: 35.20%
  • 나이지리아 승리 확률: 64.80%
